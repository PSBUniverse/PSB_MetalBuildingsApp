## Extend Derived State

```js
function computeDerived(config) {
  const halfWidth = config.width / 2
  const bays = Math.ceil(config.length / config.spacing)

  const positions = Array.from({ length: bays + 1 }, (_, i) => i * config.spacing)

  const rise = halfWidth * config.pitch

  // 👇 Lean-to derived
  const leanTo = config.leanTo?.enabled
    ? {
        ...config.leanTo,
        attachHeight: config.height,
        outerHeight: config.height - config.leanTo.drop,
        sideSign: config.leanTo.side === "right" ? 1 : -1
      }
    : null

  return {
    ...config,
    halfWidth,
    bays,
    positions,
    rise,
    leanTo
  }
}
```

# 🧱 STEP 4 — POSTS (Aligned with Main Grid)

```js
function buildLeanToPosts(ctx) {
  const group = new THREE.Group()

  const { leanTo, positions, halfWidth } = ctx
  const x = halfWidth + leanTo.width
  const signedX = leanTo.sideSign * x

  for (const z of positions) {
    const post = createPost(
      signedX,
      z,
      leanTo.outerHeight
    )
    group.add(post)
  }

  return group
}
```

👉 Uses SAME `positions` → aligned with main structure
👉 This is why it looks correct

# 🧱 STEP 5 — BEAMS

```js
function buildLeanToBeams(ctx) {
  const group = new THREE.Group()
  const { leanTo, halfWidth, length } = ctx

  const innerX = leanTo.sideSign * halfWidth
  const outerX = leanTo.sideSign * (halfWidth + leanTo.width)

  // inner beam (attached)
  group.add(createBeam(innerX, leanTo.attachHeight, length))

  // outer beam
  group.add(createBeam(outerX, leanTo.outerHeight, length))

  return group
}
```

# 🧱 STEP 6 — ROOF (THIS IS THE IMPORTANT ONE)

```js
function buildLeanToRoof(ctx) {
  const group = new THREE.Group()

  const { leanTo, length, halfWidth } = ctx

  const slopeWidth = leanTo.width
  const heightDiff = leanTo.drop

  const slopeLength = Math.sqrt(
    slopeWidth**2 + heightDiff**2
  )

  const geo = new THREE.PlaneGeometry(length, slopeLength)

  const mat = createRoofMaterial(ctx.colors.roof)

  const roof = new THREE.Mesh(geo, mat)

  roof.rotation.x = Math.PI / 2
  roof.rotation.z = leanTo.sideSign * Math.atan(heightDiff / slopeWidth)

  const baseX = leanTo.sideSign * halfWidth

  roof.position.set(
    baseX + leanTo.sideSign * slopeWidth / 2,
    leanTo.attachHeight - heightDiff / 2,
    0
  )

  group.add(roof)

  return group
}
```

# 🔗 CRITICAL ALIGNMENT RULES

These are the difference between:

* looking “right”
* looking “off”

---

## 1. SAME GRID

```js
leanTo.positions === main.positions
```

---

## 2. SAME LENGTH

Lean-to runs full length unless configured otherwise.

---

## 3. CLEAN SEAM

This must be true:

```js
leanTo.attachHeight === mainRoofEdgeHeight
```

If not:
👉 visible gap
👉 visual bug

---
